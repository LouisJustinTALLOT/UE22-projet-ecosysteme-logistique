55 def get_infos_clusters_enveloppes_convexes(k: int, df: pd.DataFrame, column_geometry: str = COLUMN_DEFAULT_GEOMETRY_NAME, is_dict: bool = False) -> pd.DataFrame:
56  """
57  Fonction permettant de récupérer des infos sur les clusters (enveloppes convexes).
58 
59  :param k: Nombre de clusters
60  :param df: La DataFrame où l'on a déjà ajouté le numéro des clusters (laissée intacte).
61  :param column_geometry: Le nom de la colonne où se situent les données géometriques (par défaut, "geometry").
62  :param is_dict: True si les paramètres sont sous forme de dictionnaire
63  :return: Une GeoDataFrame associant à chaque numéro de cluster son enveloppe convexe.
64  """
65  # Tableau numpy temporaire. Il ne sert qu'à la création de la GeoDataFrame avec les enveloppes convexes
66  temp_hulls = np.empty(k, dtype=Polygon)
67 
68  for n in range(k):
69  # Cette DataFrame contient les points du cluster numéro n
70  minidf = df.loc[df[COLUMN_CLUSTER_INDEX_NAME] == n]
71 
72  # On calcule l'enveloppe convexe
73  points = minidf.loc[:, column_geometry]
74  if is_dict:
75  points = gpd.points_from_xy(points.apply(lambda p: p["coordinates"][0]),
76  points.apply(lambda p: p["coordinates"][1]))
77 
78  multi_point = MultiPoint(np.array(points, dtype=Point))
79  else:
80  multi_point = MultiPoint(points.array)
81 
82  hull = multi_point.convex_hull
83 
84  if type(hull) == Point or type(hull) == LineString or type(hull) == GeometryCollection:
85  # S'il n'y a qu'un point dans le cluster, on ne peut pas créer de Polygon
86  temp_hulls[n] = hull
87  else:
88  temp_hulls[n] = Polygon(hull)
89 
90  return gpd.GeoDataFrame(gpd.GeoSeries(temp_hulls), columns=[COLUMN_HULLS_NAME])

139 def swap_xy(geom):
140  """
141  Inverse les coordonnées de l'objet shapely.geometry.
142  Utile pour passer objets shapely dans folium (la convention est inversée).
143  Auteur : https://gis.stackexchange.com/a/291293
144 
145  :param geom: L'objet dont on veut inverser les coordonnées (Point, Polygon, MultiPolygon, etc.)
146  :return: l'objet inversé
147  """
148  if geom.is_empty:
149  return geom
150 
151  if geom.has_z:
152  def swap_xy_coords(coords):
153  for x, y, z in coords:
154  yield (y, x, z)
155  else:
156  def swap_xy_coords(coords):
157  for x, y in coords:
158  yield (y, x)
159 
160  # Process coordinates from each supported geometry type
161  if geom.type in ('Point', 'LineString', 'LinearRing'):
162  return type(geom)(list(swap_xy_coords(geom.coords)))
163 
164  elif geom.type == 'Polygon':
165  ring = geom.exterior
166  shell = type(ring)(list(swap_xy_coords(ring.coords)))
167  holes = list(geom.interiors)
168 
169  for pos, ring in enumerate(holes):
170  holes[pos] = type(ring)(list(swap_xy_coords(ring.coords)))
171 
172  return type(geom)(shell, holes)
173 
174  elif geom.type.startswith('Multi') or geom.type == 'GeometryCollection':
175  # Recursive call
176  return type(geom)([swap_xy(part) for part in geom.geoms])
177 
178  else:
179  raise ValueError('Type %r not recognized' % geom.type)

182 def filter_nearby_paris(df: pd.DataFrame, radius: int, column_geometry: str = COLUMN_DEFAULT_GEOMETRY_NAME, is_dict: bool = False) -> pd.DataFrame:
183  """
184  Filtre les données proches du centre de Paris.
185 
186  :param df: la DataFrame à filtrer
187  :param radius: le rayon (en kilomètres)
188  :param column_geometry: la colonne où se trouvent les données géométriques (par défaut : 'geometry')
189  :return: la DataFrame filtrée
190  """
191  X = get_coords_from_object(df, column_geometry, is_dict)
192 
193  l1 = np.pi/180*PARIS_CENTER_COORDS.y
194  L1 = np.pi/180*PARIS_CENTER_COORDS.x
195  l2 = np.pi/180*X[:,1]
196  L2 = np.pi/180*X[:,0]
197 
198  distances = 20000/np.pi * (np.arccos(np.sin(l1)*np.sin(l2) + np.cos(l1)*np.cos(l2)*np.cos(L1-L2)))
199  masque = distances <= radius
200  return df[masque].reset_index(drop=True)

203 def get_coords_from_object(df: pd.DataFrame, column_geometry: str = COLUMN_DEFAULT_GEOMETRY_NAME, is_dict: bool = False) -> np.ndarray:
204  """
205  Récupère les coordonnées des points de la DataFrame.
