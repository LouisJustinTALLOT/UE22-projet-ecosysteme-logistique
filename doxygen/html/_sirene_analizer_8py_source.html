39 def swap_xy(geom:BaseGeometry):
40  """
41  Inverse les coordonnées de l'objet shapely.geometry.
42  Utile pour passer objets shapely dans folium (la convention est inversée)
43  :param geom: l'objet dont on veut inverser les coordonnées (Point, Polygon, MultiPolygon, etc.)
44  :return: l'objet inversé
45  """
46  if geom.is_empty:
47  return geom
48 
49  if geom.has_z:
50  def swap_xy_coords(coords:tuple):
51  for x, y, z in coords:
52  yield (y, x, z)
53  else:
54  def swap_xy_coords(coords:tuple):
55  for x, y in coords:
56  yield (y, x)
57 
58  # Process coordinates from each supported geometry type
59  if geom.type in ('Point', 'LineString', 'LinearRing'):
60  return type(geom)(list(swap_xy_coords(geom.coords)))
61  elif geom.type == 'Polygon':
62  ring = geom.exterior
63  shell = type(ring)(list(swap_xy_coords(ring.coords)))
64  holes = list(geom.interiors)
65  for pos, ring in enumerate(holes):
66  holes[pos] = type(ring)(list(swap_xy_coords(ring.coords)))
67  return type(geom)(shell, holes)
68  elif geom.type.startswith('Multi') or geom.type == 'GeometryCollection':
69  # Recursive call
70  return type(geom)([swap_xy(part) for part in geom.geoms])
71  else:
72  raise ValueError('Type %r not recognized' % geom.type)

213 def do_convex_hull(df:gpd.GeoDataFrame):
214  """
215  A partir de données clusterisées (le premier élément retourné par la fonction clusterize)
216  (cf. doc de la fonction clusterize), fabrique les enveloppes convexes.
217 
218  :param df: la GeoDataFrame où les clusters ont déjà été calculés par la fonction clusterize
219  :return:
220  Retourne un tuple.
221  En premier : une GeoDataFrame identique, à ceci près que l'on a ajouté la colonne 'hulls'.
222  (correspondant à un Polygon, enveloppe convexe du cluster)
223  (il se peut que ce soit un Point, si le cluster ne contient qu'un point, car les Polygons à un point sont interdits)
224  En deuxième, une GeoDataFrame plus simple avec une unique colonne 'hulls'
225  (la ligne est le numéro de cluster, et contient les enveloppes convexes)
226  """
227 
228  # Nombre de clusters
229  k = int(np.max(df['cluster']) + 1)
230 
231  hulls = np.empty(k, dtype=Polygon)
232 
233  for n in range(k):
234  # df du cluster correspondant
235  minidf = df.loc[df['cluster'] == n]
236 
237  points = minidf.loc[:, 'geometry']
238  points = gpd.points_from_xy(points.apply(lambda p: p["coordinates"][0]),
239  points.apply(lambda p: p["coordinates"][1]))
240 
241  multi_point = MultiPoint(np.array(points))
242 
243  hull = multi_point.convex_hull
244  if type(hull) == Point:
245  hulls[n] = hull
246  else:
247  hulls[n] = Polygon(hull)
248 
249  pre_df = gpd.GeoDataFrame(gpd.GeoSeries(hulls), columns=['hulls'])
250 
251  df = df.join(pre_df, how='left', on='cluster')
252 
253  return df, pre_df
