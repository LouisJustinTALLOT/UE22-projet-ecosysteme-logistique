37 def swap_xy(geom):
38  """
39  Inverse les coordonnées de l'objet shapely.geometry.
40  Utile pour passer objets shapely dans folium (la convention est inversée)
41  :param geom: l'objet dont on veut inverser les coordonnées (Point, Polygon, MultiPolygon, etc.)
42  :return: l'objet inversé
43  """
44  if geom.is_empty:
45  return geom
46 
47  if geom.has_z:
48  def swap_xy_coords(coords):
49  for x, y, z in coords:
50  yield (y, x, z)
51  else:
52  def swap_xy_coords(coords):
53  for x, y in coords:
54  yield (y, x)
55 
56  # Process coordinates from each supported geometry type
57  if geom.type in ('Point', 'LineString', 'LinearRing'):
58  return type(geom)(list(swap_xy_coords(geom.coords)))
59  elif geom.type == 'Polygon':
60  ring = geom.exterior
61  shell = type(ring)(list(swap_xy_coords(ring.coords)))
62  holes = list(geom.interiors)
63  for pos, ring in enumerate(holes):
64  holes[pos] = type(ring)(list(swap_xy_coords(ring.coords)))
65  return type(geom)(shell, holes)
66  elif geom.type.startswith('Multi') or geom.type == 'GeometryCollection':
67  # Recursive call
68  return type(geom)([swap_xy(part) for part in geom.geoms])
69  else:
70  raise ValueError('Type %r not recognized' % geom.type)

107 def clusterize(df, k):
108  """
109  Clusterise la GeoDataFrame à l'aide de la méthode des k-moyennes.
110  La colonne contenant les points doit s'appeler 'geometry'.
111  La GeoDataFrame peut contenir d'autres colonnes.
112 
113  :param df: la GeoDataFrame à clusteriser (les données doivent se trouver dans la colonne 'geometry')
114  :param k: le nombre de clusters
115  :return:
116  Retourne un tuple.
117  En premier : une GeoDataFrame identique, à ceci près que l'on a ajouté les colonnes 'cluster' et 'centroids'.
118  (correspondant au numéro du cluster, et à son centre de gravité)
119  En deuxième, une GeoDataFrame plus simple avec une unique colonne 'centroids'
120  (la ligne est le numéro de cluster, et contient le centre de gravité)
121  """
122 
123  # On commence par transformer la GeoDataFrame en tableau numpy
124  # C'est les lat / lon
125  a = pd.Series(df['geometry'].apply(lambda p: p.x))
126  b = pd.Series(df['geometry'].apply(lambda p: p.y))
127  X = np.column_stack((a, b))
128 
129  kmeans = KMeans(n_clusters=k, init='k-means++')
130  y_kmeans = kmeans.fit_predict(X)
131  k = pd.DataFrame(y_kmeans, columns=['cluster'], dtype=int)
132 
133  cluster_centers = kmeans.cluster_centers_
134  centers =gpd.points_from_xy(cluster_centers[:,0], cluster_centers[:,1])
135  centroids = pd.DataFrame(centers, columns=['centroids'])
136 
137  k = k.join(centroids, how="left", on="cluster")
138  df = df.join(k)
139  return df, centroids

142 def do_convex_hull(df):
143  """
144  A partir de données clusterisées (le premier élément retourné par la fonction clusterize)
145  (cf. doc de la fonction clusterize), fabrique les enveloppes convexes.
146 
147  :param df: la GeoDataFrame où les clusters ont déjà été calculés par la fonction clusterize
148  :return:
149  Retourne un tuple.
150  En premier : une GeoDataFrame identique, à ceci près que l'on a ajouté la colonne 'hulls'.
151  (correspondant à un Polygon, enveloppe convexe du cluster)
152  (il se peut que ce soit un Point, si le cluster ne contient qu'un point, car les Polygons à un point sont interdits)
153  En deuxième, une GeoDataFrame plus simple avec une unique colonne 'hulls'
154  (la ligne est le numéro de cluster, et contient les enveloppes convexes)
155  """
156 
157  # Nombre de clusters
158  k = int(np.max(df['cluster']) + 1)
159 
160  hulls = np.empty(k, dtype=Polygon)
161 
162  for n in range(k):
163  # df du cluster correspondant
164  minidf = df.loc[df['cluster'] == n]
165 
166  points = minidf.loc[:, 'geometry']
167 
168  multi_point = MultiPoint(points.array)
169 
170  hull = multi_point.convex_hull
171 
172  if type(hull) == Point:
173  hulls[n] = hull
174  else:
175  hulls[n] = Polygon(hull)
176 
177  pre_df = gpd.GeoDataFrame(gpd.GeoSeries(hulls), columns=['hulls'])
178 
179  df = df.join(pre_df, how='left', on='cluster')
180 
181  return df, pre_df
