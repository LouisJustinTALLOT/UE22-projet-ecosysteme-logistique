20 def swap_xy(geom):
21  """
22  Inverse les coordonnées de l'objet shapely.geometry.
23  Utile pour passer objets shapely dans folium (la convention est inversée)
24  :param geom: l'objet dont on veut inverser les coordonnées (Point, Polygon, MultiPolygon, etc.)
25  :return: l'objet inversé
26  """
27  if geom.is_empty:
28  return geom
29 
30  if geom.has_z:
31  def swap_xy_coords(coords):
32  for x, y, z in coords:
33  yield (y, x, z)
34  else:
35  def swap_xy_coords(coords):
36  for x, y in coords:
37  yield (y, x)
38 
39  # Process coordinates from each supported geometry type
40  if geom.type in ('Point', 'LineString', 'LinearRing'):
41  return type(geom)(list(swap_xy_coords(geom.coords)))
42 
43  elif geom.type == 'Polygon':
44  ring = geom.exterior
45  shell = type(ring)(list(swap_xy_coords(ring.coords)))
46  holes = list(geom.interiors)
47 
48  for pos, ring in enumerate(holes):
49  holes[pos] = type(ring)(list(swap_xy_coords(ring.coords)))
50 
51  return type(geom)(shell, holes)
52 
53  elif geom.type.startswith('Multi') or geom.type == 'GeometryCollection':
54  # Recursive call
55  return type(geom)([swap_xy(part) for part in geom.geoms])
56 
57  else:
58  raise ValueError('Type %r not recognized' % geom.type)

77 def clusterize(df, k, column_name="geometry", dict=False):
78  """
79  Clusterise à l'aide de l'algorithme des k-moyennes.
80 
81  A besoin d'une DataFrame contenant une colonne avec des Points (par défaut, son nom est geometry)
82  Indiquer dict=True si jamais la colonne contenant les Points ne contient
83  pas l'objet Point (module shapely), mais un dictionnaire
84  (quand ça vient pas d'un GeoJSON en général)
85 
86  Attention, fait du en-place.
87  Retourne trois choses :
88  - la df rentrée, avec trois colonnes en plus:
89  "cluster" (numéro du cluster), "centroids" (centre de masse correspondant)
90  et "hulls" (enveloppe convexe)
91  Ainsi, pour chaque point, on connaît les infos du cluster auquel il a été affecté
92  - les centres de masse et les enveloppes convexes de chaque cluster (concrètement, une GeoDataFrame
93  avec comme index le numéro du cluster et comme colonnes les centres de masses et les enveloppes convexes)
94 
95  """
96 
97  #===========================================================
98  # Commençons par faire le clustering et récupérer les centres
99  # ==========================================================
100  kmeans = KMeans(n_clusters=k, random_state=0)
101 
102  # Ceci contient des coordonnées (x, y) des points
103  X = None
104 
105  if dict:
106  a = pd.Series(df['geometry'].apply(lambda p: p["coordinates"][0]))
107  b = pd.Series(df['geometry'].apply(lambda p: p["coordinates"][1]))
108 
109  X = np.column_stack((a, b))
110  else:
111  a = pd.Series(df[column_name].apply(lambda p: p.x))
112  b = pd.Series(df[column_name].apply(lambda p: p.y))
113 
114  X = np.column_stack((a, b))
115 
116  # Cette DataFrame associe à chaque indice de ligne (ie à chaque point)
117  # son cluster, dans la colonne "cluster"
118  clusters = gpd.GeoDataFrame(kmeans.fit_predict(X), columns=['cluster'], dtype=int)
119 
120  # Cette DataFrame associe à chaque numéro de cluster le centre de masse
121  # correspondant, dans la colonne "centroids"
122  centroids = pd.DataFrame(gpd.points_from_xy(kmeans.cluster_centers_[:, 0],
123  kmeans.cluster_centers_[:, 1]),
124  columns=['centroids']
125  )
126 
127  # Désormais, on a ajouté à cette DataFrame une colonne centroids.
128  # Chaque point est donc associé à son cluster et le centre de masse correspondant
129  clusters = clusters.join(centroids, how="left", on="cluster")
130 
131  # La DataFrame originale contient désormais deux colonnes en plus :
132  # - "cluster" : le numéro du cluster
133  # - "centroids": le centre de masse correspondant
134  df = df.join(clusters)
135 
136  # ================================================================
137  # Désormais, on récupère les enveloppes convexes de chaque cluster
138  # ================================================================
139 
140  # Tableau numpy temporaire. Il ne sert qu'à la création de la GeoDataFrame avec les enveloppes convexes
141  temp_hulls = np.empty(k, dtype=Polygon)
142 
143  for n in range(k):
144  # Cette DataFrame contient les points du cluster numéro n
145  minidf = df.loc[df['cluster'] == n]
146 
147  # On calcule l'enveloppe convexe
148  points = minidf.loc[:, column_name]
149  if dict:
150  points = gpd.points_from_xy(points.apply(lambda p: p["coordinates"][0]),
151  points.apply(lambda p: p["coordinates"][1]))
152 
153  multi_point = MultiPoint(np.array(points, dtype=Point))
154  else:
155  multi_point = MultiPoint(points.array)
156 
157  hull = multi_point.convex_hull
158 
159  if type(hull) == Point:
160  # S'il n'y a qu'un point dans le cluster, on ne peut pas créer de Polygon
161  temp_hulls[n] = hull
162  else:
163  temp_hulls[n] = Polygon(hull)
164 
165  hulls = gpd.GeoDataFrame(gpd.GeoSeries(temp_hulls), columns=['hulls'])
166  df = df.join(hulls, how='left', on='cluster')
167 
168  df_clusters = centroids.join(hulls)
169

171  # RESULTATS
172  # df contient trois colonnes en plus :
173  # "cluster" (n° cluster),
174  # "centroids" (centres) ,
175  # "hulls"
176  # df_clusters associe à chaque numéro de cluster son centre et son enveloppe convexe
177  # ==================================================
178
